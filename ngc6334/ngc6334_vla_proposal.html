def setup_axes(header,myfig):
    ax = pywcsgrid2.subplot(111, header=header)

    divider = make_axes_locatable(ax)

    myfig.add_axes(cax)

    return ax, cax

def coorlocate(f,ra_str,dec_str):
    pC = SkyCoord(ra_str, dec_str, frame='fk5')  
    print pC.to_string().split(" ")[0]
    ra = float(pC.to_string().split(" ")[0])
    dec = float(pC.to_string().split(" ")[1])
    return f.world2pixel(ra,dec)

def pointings(ax, f, ra = '', dec = ''):
    ax.annotate("+", coorlocate(f,ra,  dec), size=12, ha="center", va="center",color="red")
    el = Ellipse(xy=coorlocate(f,ra,  dec), width=1.3*60/0.2, height=1.3*60/0.2, angle=0,facecolor = 'None', edgecolor = 'grey')
    ax.add_artist(el)
    return 0
