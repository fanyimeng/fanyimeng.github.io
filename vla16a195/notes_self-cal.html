Prepare the tools

  • Split .ms:
def splitInitial(inputMS, outputMS):
    '''
    Split a measurement set (MS) to create a new one. This is for the first 
    split, which split out the data column 'data', because there is, probably, 
    no column 'corrected' in the MS.  
    '''
    os.system("rm -rf " + outputMS)
    split(
        vis = inputMS,
        outputvis = outputMS,
        datacolumn = "data"
        )
    return 0

def splitCorrected(inputMS, outputMS):
    '''
    Split a measurement set (MS) to create a new one, using only the 
    data column `corrected`
    '''
    os.system("rm -rf " + outputMS)
    split(
        vis = inputMS,
        outputvis = outputMS,
        datacolumn = "corrected"
        )
    return 0
  • Generate phase calibration table
def GeneratePhaseCal(myvis, mycaltable, mysolint, myminsnr):
  gaincal(
            vis=myvis, 
            caltable=mycaltable,
            field='',
            spw='',
            selectdata=False,
            solint=mysolint, 
            refant='ea04',
            minblperant=4,
            minsnr= myminsnr, 
            gaintype='G',
            calmode='p',
            append=False
            )
  return 0
  • Apply the phase calibration table
def ApplyPhaseCal(myvis, mycaltable):
  applycal(
             vis=myvis,
             field='',
             spw='',
             selectdata=False,
             gaintable= [mycaltable],
             gainfield=[''],
             interp=['nearest'],
             calwt=[False],
             applymode='calflag'
             )
  return 0
  • Plot cal solutions (not used in work flow)
def plotPhaseCal(myband,mycase):
    plotcal(
            caltable="%s_spw2-33_%s.selfcal"%(myband,mycase),
            xaxis="time",
            yaxis="phase",
            subplot=421,
            iteration="antenna",
            plotrange=[0,0,-20,20],
            markersize=1,
            fontsize=7.0,
            figfile="%s_spw2-33_%s.selfcal.png"%(myband,mycase)
            )
    return 0

def coarseImaging(mycase,myvis,myniter,mygain,myimsize,mycell):
  from time import gmtime, strftime
  myweighting = "natural"
  mypsfmode = "hogbom"
  mythreshold = '0.015mJy'
  myphasecenter='J2000 17h47m19.75 -28d23m07'
  print"=============="
  print "Started: "+strftime("%Y-%m-%d %H:%M:%S")
  print "Case: "+mycase
  print "Vis: "+myvis
  print"=============="
  clean(
        vis        = myvis,
        imagename  = mycase,
        mode       = 'mfs',
        niter      = myniter,
        gain       = mygain, 
        threshold  = mythreshold,
        psfmode    = mypsfmode,
        multiscale = [0], 
        imsize     = myimsize, 
        cell       = mycell,
        phasecenter= myphasecenter,
        weighting  = myweighting,
        usescratch = True,
        interactive= False,
        imagermode = 'mosaic',
        )
  print"=============="
  print "Finished: "+strftime("%Y-%m-%d %H:%M:%S")
  print "Case: "+mycase
  print "Vis: "+myvis
  print"=============="
  return 0
